if ($.cookie('alert') != "true") {
        $(".promo_banner").show();
        $(".promo_banner-close").click(function() {
            $(this).parent().slideUp();
            
            var date = new Date();
            date.setTime(date.getTime() + 24 * 60 * 60 * 1000);
            $.cookie('alert', "true", {
                expires: date
            });
        });
    }
    else {
        $(".promo_banner").remove();
    }
